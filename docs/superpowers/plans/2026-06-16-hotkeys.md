# Hotkeys Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add LDiff keyboard shortcuts through a frontend action registry shared by native Tauri menu accelerators and in-window keydown handling.

**Architecture:** The frontend owns command semantics, enabled checks, focus checks, and blocked messages. Tauri native menu items are a thin adapter that emits stable action ids to the webview; DOM shortcuts are another adapter that resolves keydown events to the same action ids. Existing `App.tsx` workflow handlers remain the only place that performs file open, save, search, tab, and merge behavior.

**Tech Stack:** React 19, TypeScript, Vitest, Testing Library, Playwright render verifier, Tauri 2.11.2 Rust menu APIs, shadcn/ui, Monaco.

```ts
import type { AppActionId } from "@/lib/actions";

export type PlatformName = "darwin" | "linux" | "win32" | string;

export interface KeyboardLikeEvent {
  key: string;
  metaKey: boolean;
  ctrlKey: boolean;
  altKey: boolean;
  shiftKey: boolean;
  target?: EventTarget | null;
}

export interface ShortcutBinding {
  actionId: AppActionId;
  shortcut: string;
}

export interface ParsedShortcut {
  key: string;
  cmdOrCtrl: boolean;
  shift: boolean;
  alt: boolean;
}

export type FocusKind = "none" | "editable";

export function parseShortcut(shortcut: string): ParsedShortcut {
  const parts = shortcut.split("+").map((part) => part.trim()).filter(Boolean);
  const key = parts.at(-1);
  if (!key) throw new Error(`invalid shortcut: ${shortcut}`);
  return {
    key: normalizeKey(key),
    cmdOrCtrl: parts.slice(0, -1).includes("CmdOrCtrl"),
    shift: parts.slice(0, -1).includes("Shift"),
    alt: parts.slice(0, -1).includes("Alt"),
  };
}

export function shortcutMatches(event: KeyboardLikeEvent, shortcut: ParsedShortcut, platform: PlatformName): boolean {
  const expectedCommand = platform === "darwin" ? event.metaKey : event.ctrlKey;
  if (shortcut.cmdOrCtrl !== expectedCommand) return false;
  if (!shortcut.cmdOrCtrl && (event.metaKey || event.ctrlKey)) return false;
  if (shortcut.shift !== event.shiftKey) return false;
  if (shortcut.alt !== event.altKey) return false;
  return normalizeKey(event.key) === shortcut.key;
}

export function matchShortcut(
  event: KeyboardLikeEvent,
  bindings: ShortcutBinding[],
  platform: PlatformName = currentPlatform(),
): AppActionId | undefined {
  return bindings.find((binding) => shortcutMatches(event, parseShortcut(binding.shortcut), platform))?.actionId;
}

export function classifyFocusTarget(target: EventTarget | null | undefined): FocusKind {
  if (!(target instanceof Element)) return "none";
  if (target.closest(".monaco-editor")) return "editable";
  if (target.closest("[contenteditable='true']")) return "editable";
  if (target instanceof HTMLTextAreaElement) return "editable";
  if (target instanceof HTMLInputElement) {
    const nonText = new Set(["button", "checkbox", "color", "file", "radio", "range", "reset", "submit"]);
    return nonText.has(target.type) ? "none" : "editable";
  }
  return "none";
}

export function currentPlatform(): PlatformName {
  const platform = globalThis.navigator?.platform?.toLowerCase() ?? "";
  if (platform.includes("mac")) return "darwin";
  if (platform.includes("win")) return "win32";
  return "linux";
}

function normalizeKey(key: string): string {
  if (key === " ") return "space";
  if (key.length === 1) return key.toLowerCase();
  return key.toLowerCase();
}
```

import type { FocusKind, ShortcutBinding } from "@/lib/shortcuts";
import type { Mode, Side } from "@/lib/types";

export type AppActionId =
  | "file.openLeft"
  | "file.openRight"
  | "file.refresh"
  | "file.save"
  | "edit.clearStaged"
  | "search.toggle"
  | "search.runContextual"
  | "view.togglePreferences"
  | "workspace.focusFiles"
  | "workspace.nextTab"
  | "workspace.previousTab"
  | "workspace.closeTab"
  | "merge.copyToLeft"
  | "merge.copyToRight"
  | "merge.takeAllToLeft"
  | "merge.takeAllToRight"
  | "merge.moveHunkToLeft"
  | "merge.moveHunkToRight";

export interface AppActionDefinition {
  id: AppActionId;
  label: string;
  group: "File" | "Edit" | "Search" | "View" | "Workspace" | "Merge";
  shortcut: string;
  contentChanging?: boolean;
}

export interface AppActionContext {
  mode: Mode;
  activeTab: "files" | string;
  openTabs: string[];
  selectedPath?: string;
  selectedCanCopyLeft: boolean;
  selectedCanCopyRight: boolean;
  stagedTarget?: Side;
  stagedCount: number;
  hunkMerge: boolean;
  focusKind: FocusKind;
}

export interface AppActionHandlers {
  openLeft: () => void | Promise<void>;
  openRight: () => void | Promise<void>;
  refresh: () => void | Promise<void>;
  save: () => void | Promise<void>;
  clearStaged: () => void | Promise<void>;
  toggleSearch: () => void | Promise<void>;
  runContextualSearch: () => void | Promise<void>;
  togglePreferences: () => void | Promise<void>;
  focusFiles: () => void | Promise<void>;
  nextTab: () => void | Promise<void>;
  previousTab: () => void | Promise<void>;
  closeActiveTab: () => void | Promise<void>;
  copyToLeft: () => void | Promise<void>;
  copyToRight: () => void | Promise<void>;
  takeAllToLeft: () => void | Promise<void>;
  takeAllToRight: () => void | Promise<void>;
  moveHunkToLeft: () => void | Promise<void>;
  moveHunkToRight: () => void | Promise<void>;
  reportBlocked: (message: string) => void;
}

export interface AppActionState {
  enabled: boolean;
  blockedReason?: string;
}

export const ACTION_DEFINITIONS: AppActionDefinition[] = [
  { id: "file.openLeft", label: "Open Left Source", group: "File", shortcut: "CmdOrCtrl+O" },
  { id: "file.openRight", label: "Open Right Target", group: "File", shortcut: "CmdOrCtrl+Shift+O" },
  { id: "file.refresh", label: "Refresh Sources", group: "File", shortcut: "CmdOrCtrl+R" },
  { id: "file.save", label: "Save Staged Target", group: "File", shortcut: "CmdOrCtrl+S" },
  { id: "edit.clearStaged", label: "Clear Staged Changes", group: "Edit", shortcut: "CmdOrCtrl+Shift+Backspace", contentChanging: true },
  { id: "search.toggle", label: "Toggle Search", group: "Search", shortcut: "CmdOrCtrl+F" },
  { id: "search.runContextual", label: "Run Search Or Find", group: "Search", shortcut: "CmdOrCtrl+Enter" },
  { id: "view.togglePreferences", label: "Toggle Preferences", group: "View", shortcut: "CmdOrCtrl+," },
  { id: "workspace.focusFiles", label: "Focus Files", group: "Workspace", shortcut: "CmdOrCtrl+1" },
  { id: "workspace.nextTab", label: "Next Tab", group: "Workspace", shortcut: "CmdOrCtrl+Tab" },
  { id: "workspace.previousTab", label: "Previous Tab", group: "Workspace", shortcut: "CmdOrCtrl+Shift+Tab" },
  { id: "workspace.closeTab", label: "Close Active Tab", group: "Workspace", shortcut: "CmdOrCtrl+W" },
  { id: "merge.copyToLeft", label: "Copy Entry To Left", group: "Merge", shortcut: "Alt+[", contentChanging: true },
  { id: "merge.copyToRight", label: "Copy Entry To Right", group: "Merge", shortcut: "Alt+]", contentChanging: true },
  { id: "merge.takeAllToLeft", label: "Take All Into Left", group: "Merge", shortcut: "Alt+Shift+[", contentChanging: true },
  { id: "merge.takeAllToRight", label: "Take All Into Right", group: "Merge", shortcut: "Alt+Shift+]", contentChanging: true },
  { id: "merge.moveHunkToLeft", label: "Move Hunk Into Left", group: "Merge", shortcut: "CmdOrCtrl+Alt+[", contentChanging: true },
  { id: "merge.moveHunkToRight", label: "Move Hunk Into Right", group: "Merge", shortcut: "CmdOrCtrl+Alt+]", contentChanging: true },
];

export function shortcutBindings(): ShortcutBinding[] {
  return ACTION_DEFINITIONS.map(({ id, shortcut }) => ({ actionId: id, shortcut }));
}

export function isAppActionId(value: string): value is AppActionId {
  return ACTION_DEFINITIONS.some((definition) => definition.id === value);
}

export function getActionState(actionId: AppActionId, context: AppActionContext): AppActionState {
  const definition = ACTION_DEFINITIONS.find((candidate) => candidate.id === actionId);
  if (!definition) return { enabled: false, blockedReason: "Unknown command." };
  if (definition.contentChanging && context.focusKind === "editable") {
    return { enabled: false, blockedReason: "Finish editing or leave the editor before running this merge shortcut." };
  }
  switch (actionId) {
    case "file.openRight":
      return context.mode === "compare"
        ? { enabled: true }
        : { enabled: false, blockedReason: "Open right source is available only in Compare mode." };
    case "file.save":
      return context.stagedTarget && context.stagedCount > 0
        ? { enabled: true }
        : { enabled: false, blockedReason: "No staged changes to save." };
    case "edit.clearStaged":
      return context.stagedCount > 0
        ? { enabled: true }
        : { enabled: false, blockedReason: "No staged changes to clear." };
    case "workspace.nextTab":
    case "workspace.previousTab":
      return context.openTabs.length > 0
        ? { enabled: true }
        : { enabled: false, blockedReason: "Open a diff tab before switching tabs." };
    case "workspace.closeTab":
      return context.activeTab === "files"
        ? { enabled: false, blockedReason: "Open a diff tab before closing a tab." }
        : { enabled: true };
    case "merge.copyToLeft":
      return context.selectedCanCopyLeft
        ? { enabled: true }
        : { enabled: false, blockedReason: "Select an entry before copying to the left." };
    case "merge.copyToRight":
      return context.selectedCanCopyRight
        ? { enabled: true }
        : { enabled: false, blockedReason: "Select an entry before copying to the right." };
    case "merge.takeAllToLeft":
    case "merge.takeAllToRight":
    case "merge.moveHunkToLeft":
    case "merge.moveHunkToRight":
      return context.hunkMerge
        ? { enabled: true }
        : { enabled: false, blockedReason: "Open a diff tab before moving hunks." };
    default:
      return { enabled: true };
  }
}

export async function dispatchAppAction(
  actionId: AppActionId,
  context: AppActionContext,
  handlers: AppActionHandlers,
): Promise<boolean> {
  const state = getActionState(actionId, context);
  if (!state.enabled) {
    handlers.reportBlocked(state.blockedReason ?? "Command is not available.");
    return false;
  }
  const actionHandlers: Record<AppActionId, () => void | Promise<void>> = {
    "file.openLeft": handlers.openLeft,
    "file.openRight": handlers.openRight,
    "file.refresh": handlers.refresh,
    "file.save": handlers.save,
    "edit.clearStaged": handlers.clearStaged,
    "search.toggle": handlers.toggleSearch,
    "search.runContextual": handlers.runContextualSearch,
    "view.togglePreferences": handlers.togglePreferences,
    "workspace.focusFiles": handlers.focusFiles,
    "workspace.nextTab": handlers.nextTab,
    "workspace.previousTab": handlers.previousTab,
    "workspace.closeTab": handlers.closeActiveTab,
    "merge.copyToLeft": handlers.copyToLeft,
    "merge.copyToRight": handlers.copyToRight,
    "merge.takeAllToLeft": handlers.takeAllToLeft,
    "merge.takeAllToRight": handlers.takeAllToRight,
    "merge.moveHunkToLeft": handlers.moveHunkToLeft,
    "merge.moveHunkToRight": handlers.moveHunkToRight,
  };
  await actionHandlers[actionId]();
  return true;
}

    if (openTabs.length === 0) return;
    if (activeTab === "files") {
      const target = direction > 0 ? openTabs[0] : openTabs.at(-1);
      if (target) focusTab(target.path);
      return;

    const index = openTabs.findIndex((tab) => tab.path === activeTab);
    const nextIndex = index < 0 ? 0 : (index + direction + openTabs.length) % openTabs.length;
    focusTab(openTabs[nextIndex].path);

fn install_app_menu(app: &tauri::App) -> tauri::Result<()> {
    let handle = app.handle();
    let menu = Menu::new(handle)?;
    for group in ["File", "Edit", "Search", "View", "Workspace", "Merge"] {
        let mut submenu = SubmenuBuilder::new(handle, group);
        for (_, action_id, label, shortcut) in MENU_ACTIONS.iter().filter(|(candidate, _, _, _)| *candidate == group) {
            let item = MenuItemBuilder::with_id(*action_id, *label)
                .accelerator(*shortcut)
                .build(handle)?;
            submenu = submenu.item(&item);
        }
        menu.append(&submenu.build()?)?;
    }
    app.set_menu(menu)?;
    Ok(())
}
